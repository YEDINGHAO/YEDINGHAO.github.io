void solve() {
    int n,m,k;
    cin>>n>>m>>k;
    int minn=min(n,m);
    int maxx=max(n,m);
    if(maxx<k){
        cout<<-1<<'\n';return ;
    }
    maxx-=k;
    if(maxx>minn){
        cout<<-1<<'\n';return ;
    }
    else if(minn-maxx>k){
        cout<<-1<<'\n';return ;
    }
    if(n>m){
        rep(i,1,k)cout<<'0';
        rep(i,1,maxx)cout<<"10";
        rep(i,1,minn-maxx)cout<<'1';
    }else{
        rep(i,1,k)cout<<'1';
        rep(i,1,maxx)cout<<"01";
        rep(i,1,minn-maxx)cout<<'0';
    }cout<<'\n';
}
